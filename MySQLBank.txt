Create Table BankManagementSystem( 
customerAccountNumber int(8) Primary Key, 
customerName varchar(30) Not Null,
customerAccountBalance int(8) Not Null,
customerAddress varchar(30) Not Null,
customerMobileNumber varchar(10) Not Null,
customerEmailID varchar(20) ,
customerDebitCardNo varchar(16)  Not Null,
customerATMPin int (4)  Not Null,
branchifscCode varchar(6) Not Null,
branchName varchar(10) Not Null  
);

( 1 ) customerAccountNumber
( 2 ) customerName 
( 3 ) customerAccountBalance
( 4 ) customerAddress 
( 5 ) customerMobileNumber 
( 6 ) customerEmailID 
( 7 ) customerDebitCardNo 
( 8 ) customerATMPin 
( 9 ) branchifscCode 
( 10 ) branchName
